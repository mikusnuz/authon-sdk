Quick Start
import authon "github.com/mikusnuz/authon-sdk/go"

backend := authon.NewAuthonBackend(
    os.Getenv("AUTHON_API_KEY"),
    authon.WithAPIURL(os.Getenv("AUTHON_API_URL")),
)

// Verify token from cookie
user, err := backend.VerifyToken(ctx, token)

// List users
users, err := backend.Users.List(ctx, &authon.ListOptions{Page: 1, PerPage: 20})
